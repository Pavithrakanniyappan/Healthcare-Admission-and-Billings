import numpy as np
def replace_outliers(df,column):
    Q1=df[column].quantile(0.25)   
    Q3=df[column].quantile(0.75)

    IQR=Q3-Q1

    upper_bound=Q3+1.5*IQR  
    lower_bound=Q1-1.5*IQR

    # capping-replacing all higher values with upper bound (75% percentile)
    df[column]=np.where(df[column]>upper_bound,upper_bound,df[column])      # capping
    # flooring -replace all lower values with lower bound (25% percentile)
    df[column]=np.where(df[column]<lower_bound,lower_bound,df[column])    # flooring
replace_outliers(df,"Visa_Fee_USD")
